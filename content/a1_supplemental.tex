GraphQL Abfrage um alle Episoden des Podcasts "Radio Wissen" zu erhalten
Die Programset-ID verweist auf den speziellen Podcast.

\begin{verbatim}
{
    programSet(id: 5945518) {
        title
        items(
            orderBy: PUBLISH_DATE_DESC
            filter: {
            isPublished: {
                equalTo: true
            }
            }
        ) {
            
            nodes {
                title,
                audios {
                    downloadUrl
                }
            }
        }
    }
}

\end{verbatim}
